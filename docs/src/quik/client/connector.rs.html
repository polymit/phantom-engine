14/// Represents an established HTTP/2 connection with a fixed Chrome identity.
15///
16/// This structure holds the active H2 request handle and the profile used
17/// to establish the connection, ensuring that all requests on this stream
18/// adhere to the same behavioral constraints.
19pub struct QuikConnection {
20    /// The handle used to initiate new H2 streams.
21    pub h2: SendRequest<Bytes>,
22    /// The profile used for TLS and H2 handshake parity.
23    pub profile: ChromeProfile,
24}
25
26/// Establishes a new network connection following the Chrome transport pipeline.
27///
28/// 1. **Proxy/TCP**: Dials the target host (optionally via a SOCKS5/HTTP tunnel).
29/// 2. **TLS Handshake**: Performs a BoringSSL handshake with ClientHello permutation.
30/// 3. **ALPS/ECH**: Injects per-connection application settings and ECH GREASE.
31/// 4. **H2 Handshake**: Negotiates the HTTP/2 session with Chrome-identical SETTINGS.
32pub async fn connect(
33    host: &str,
34    port: u16,
35    addr: SocketAddr,
36    profile: &ChromeProfile,
37    proxy: Option<&Proxy>,
38) -> Result<QuikConnection> {
39    // Stage 1: Establish raw TCP transport.
40    let tcp = if let Some(p) = proxy {
41        dial_proxy(p, host, port).await?
42    } else {
43        TcpStream::connect(addr).await?
44    };
45
46    // Stage 2: Configure the TLS connector.
47    let connector = build_connector(&profile.tls)?;
48    let mut config = connector.configure()?;
49
50    // Request OCSP stapling to match Chrome's certificate verification behavior.
51    config.set_status_type(boring::ssl::StatusType::OCSP)?;
52
53    // Stage 3: Per-connection FFI for advanced Chrome features.
54    let ssl_ptr = config.as_ptr();
55
56    // SAFETY: The `ssl_ptr` is valid for the duration of the configuration phase.
57    // We pass valid pointers for the ALPN protocol "h2" and the static ALPS buffer.
58    unsafe {
59        if profile.tls.enable_ech_grease {
60            boring_sys::SSL_set_enable_ech_grease(ssl_ptr, 1);
61        }
62        if profile.tls.alps_enabled {
63            // Chrome 134 ALPS H2 settings:
64            // ID 1: 65536, ID 2: 0, ID 4: 6291456, ID 6: 262144
65            let alps_data: [u8; 24] = [
66                0, 1, 0, 1, 0, 0,
67                0, 2, 0, 0, 0, 0,
68                0, 4, 0, 96, 0, 0,
69                0, 6, 0, 4, 0, 0,
70            ];
71
72            boring_sys::SSL_add_application_settings(
73                ssl_ptr,
74                b"h2".as_ptr(),
75                2,
76                alps_data.as_ptr(),
77                alps_data.len(),
78            );
79        }
80    }
81
82    // Stage 4: TLS handshake.
83    let tls_stream = tokio_boring::connect(config, host, tcp)
84        .await
85        .map_err(|e| {
86            tracing::error!("TLS handshake failed: {:?}", e);
87            e
88        })?;
89
90    // Stage 5: HTTP/2 handshake.
91    let mut h2_builder = http2::client::Builder::new();
92    configure_builder(&mut h2_builder, &profile.h2);
93
94    let (h2, connection) = h2_builder.handshake(tls_stream).await?;
95
96    // Drive the connection in the background. If this task terminates,
97    // the H2 session is considered dead.
98    tokio::spawn(async move {
99        if let Err(e) = connection.await {
100            tracing::error!("HTTP/2 connection driver failed: {:?}", e);
101        }
102    });
103
104    Ok(QuikConnection {
105        h2,
106        profile: profile.clone(),
107    })
108}

110impl QuikConnection {
111    /// Dispatches an HTTP request over the established H2 session.
112    pub async fn send(
113        &mut self,
114        request: http::Request<()>,
115        body: Option<Bytes>,
116    ) -> Result<Response> {
117        let url_str = request.uri().to_string();
118        if let Some(data) = body {
119            let (response_future, mut send_stream) = self.h2.send_request(request, false)?;
120            send_stream.send_data(data, true)?;
121            let response = response_future.await?;
122            Ok(Response::new(response, url_str))
123        } else {
124            let (response_future, _) = self.h2.send_request(request, true)?;
125            let response = response_future.await?;
126            Ok(Response::new(response, url_str))
127        }
128    }
129}
