1use crate::profile::ChromeProfile;
2use http::header::{HeaderMap, HeaderValue, ACCEPT, ACCEPT_ENCODING, ACCEPT_LANGUAGE, USER_AGENT};
3
4/// Injects Chrome 134-identical headers into the provided request map.
5///
6/// This function enforces the exact header sequence, Client Hint values,
7/// and security metadata required to pass network-layer identity checks.
8///
9/// ## Navigation Metadata
10/// - **Sec-Fetch-***: Injects mode, site, user, and dest headers based on
11///   the current navigation state.
12/// - **Client Hints**: Populates `sec-ch-ua` brands and platform strings.
13/// - **HPACK Sensitivity**: Marks `cookie` and `authorization` headers as
14///   sensitive to prevent them from being indexed in the HPACK dynamic table,
15///   matching Chromium's security behavior.
16pub fn inject_chrome_headers(
17    headers: &mut HeaderMap,
18    profile: &ChromeProfile,
19    sec_fetch_site: &str,
20    is_initial_navigation: bool,
21) {
22    // 1. Client Hints (Sec-CH-UA)
23    // These headers provide granular version and platform information to the server.
24    if let Ok(val) = HeaderValue::from_str(&profile.headers.sec_ch_ua) {
25        headers.insert("sec-ch-ua", val);
26    }
27    headers.insert("sec-ch-ua-mobile", HeaderValue::from_static("?0"));
28    if let Ok(val) = HeaderValue::from_str(&profile.headers.sec_ch_ua_platform) {
29        headers.insert("sec-ch-ua-platform", val);
30    }
31
32    // 2. Navigation / Fetch metadata
33    headers.insert("upgrade-insecure-requests", HeaderValue::from_static("1"));
34    if let Ok(val) = HeaderValue::from_str(&profile.headers.user_agent) {
35        headers.insert(USER_AGENT, val);
36    }
37    
38    // Exact Chrome 134 Accept string including avif, webp, and signed-exchange.
39    headers.insert(ACCEPT, HeaderValue::from_static("text/html,application/xhtml+xml,application/xml;q=0.9,image/avif,image/webp,image/apng,*/*;q=0.8,application/signed-exchange;v=b3;q=0.7"));
40
41    // Inject dynamic sec-fetch-* state based on the current redirect context.
42    if let Ok(val) = HeaderValue::from_str(sec_fetch_site) {
43        headers.insert("sec-fetch-site", val);
44    }
45    headers.insert("sec-fetch-mode", HeaderValue::from_static("navigate"));
46
47    // The 'sec-fetch-user' header is present ONLY on the first hop of a user-initiated navigation.
48    if is_initial_navigation {
49        headers.insert("sec-fetch-user", HeaderValue::from_static("?1"));
50    }
51
52    headers.insert("sec-fetch-dest", HeaderValue::from_static("document"));
53
54    // 3. Compression & Language
55    let encoding = if profile.headers.zstd_encoding {
56        "gzip, deflate, br, zstd"
57    } else {
58        "gzip, deflate, br"
59    };
60    headers.insert(ACCEPT_ENCODING, HeaderValue::from_static(encoding));
61    headers.insert(ACCEPT_LANGUAGE, HeaderValue::from_static("en-US,en;q=0.9"));
62
63    // 4. Chrome Priority Header (u=0, i for navigations)
64    if profile.headers.include_priority_header {
65        headers.insert("priority", HeaderValue::from_static("u=0, i"));
66    }
67
68    // 5. HPACK "Never Index" (Sensitive) markers.
69    // Chrome explicitly marks cookies and auth headers as sensitive. This forces
70    // the HPACK encoder to use the "Literal Header Field Never Indexed" representation,
71    // which prevents these values from entering the dynamic table (CRIME mitigation).
72    for (name, value) in headers.iter_mut() {
73        if name == "cookie" || name == "authorization" {
74            value.set_sensitive(true);
75        }
76    }
77}
