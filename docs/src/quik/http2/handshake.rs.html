7/// Configures an HTTP/2 client builder with parameters that replicate a real Chrome 134 handshake.
8///
9/// Standard HTTP/2 libraries often use defaults that are trivial to detect (e.g., ascending
10/// SETTINGS IDs or `:method :scheme :path :authority` pseudo-header order). This function
11/// overrides those defaults to match Chromium's specialized transport behavior.
12///
13/// ## Key Enforcements
14/// - **SETTINGS Order**: Explicitly set to [1, 2, 4, 6] (Header Table, Push, Window Size, Max Header List).
15/// - **Absence of IDs 3/5**: Chrome does not send `MAX_CONCURRENT_STREAMS` or `MAX_FRAME_SIZE` in its initial settings.
16/// - **Pseudo-header Sequence**: Reorders pseudo-headers to `m,a,s,p` (Method, Authority, Scheme, Path).
17/// - **Priority Signaling**: Embeds a priority block in the initial HEADERS frame.
18pub fn configure_builder(builder: &mut Builder, profile: &Http2Profile) {
19    // 1. SETTINGS Frame Order [1, 2, 4, 6]
20    // The sequence of these IDs is a high-entropy fingerprint signal used by WAFs.
21    let mut settings_order = SettingsOrder::builder();
22    settings_order = settings_order.push(SettingId::HeaderTableSize); // 1
23    settings_order = settings_order.push(SettingId::EnablePush); // 2
24    settings_order = settings_order.push(SettingId::InitialWindowSize); // 4
25    settings_order = settings_order.push(SettingId::MaxHeaderListSize); // 6
26    builder.settings_order(settings_order.build());
27
28    // 2. SETTINGS Frame Values
29    builder.header_table_size(profile.settings.header_table_size);
30    builder.enable_push(profile.settings.enable_push);
31    builder.initial_window_size(profile.settings.initial_window_size);
32    builder.max_header_list_size(profile.settings.max_header_list_size);
33
34    // 3. Connection-Level Window Update
35    // Chrome immediately expands its connection window beyond the RFC default.
36    builder.initial_connection_window_size(profile.initial_connection_window_size);
37
38    // 4. Pseudo-header Sequence (m,a,s,p)
39    // Moving `:authority` to the second position is the most recognizable Chrome H2 marker.
40    let mut pseudo_order = PseudoOrder::builder();
41    for id in &profile.pseudo_order {
42        match id {
43            QuikPseudoOrder::Method => {
44                pseudo_order = pseudo_order.push(PseudoId::Method);
45            }
46            QuikPseudoOrder::Authority => {
47                pseudo_order = pseudo_order.push(PseudoId::Authority);
48            }
49            QuikPseudoOrder::Scheme => {
50                pseudo_order = pseudo_order.push(PseudoId::Scheme);
51            }
52            QuikPseudoOrder::Path => {
53                pseudo_order = pseudo_order.push(PseudoId::Path);
54            }
55        }
56    }
57    builder.headers_pseudo_order(pseudo_order.build());
58
59    // 5. HEADERS Priority Block
60    // Chrome embeds priority metadata (dep=0, weight=256, exclusive=true) inside
61    // the HEADERS frame rather than sending a separate PRIORITY frame.
62    builder.headers_stream_dependency(StreamDependency::new(
63        StreamId::ZERO,
64        profile.headers_priority.weight,
65        profile.headers_priority.exclusive,
66    ));
67}
